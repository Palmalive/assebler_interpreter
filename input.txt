//тестовая программа на ассемблере
mov RDX, 10          // помещаем в RDX число 10 - 1010 в двоичной системе
mov RAX, 3           // помещаем в RAX число 3 - 0011 в двоичной системе
and RBX, RDX, RAX      // RBX = RDX & RAX = 1010 & 0011 = 0010 = 2
orr RCX, RDX, RAX      // RCX = RDX | RAX = 1010 | 0011 = 1011 = 11
